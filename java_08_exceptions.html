Exception Handling

An exception is an unexpected runtime event that causes abnormal termination. Java's exception handling mechanism lets you catch and recover from these events gracefully, making Java a Robust language.

🛑

Error

JVM-level problem (OutOfMemory, StackOverflow). Cannot be handled programmatically.

⚠️

Exception

Recoverable runtime problem. Can be caught and handled with try-catch.

📋

Checked

Checked at compile time — IOException, SQLException. Must use try-catch or throws.

💥

Unchecked

Runtime exceptions — ArithmeticException, NullPointerException, ArrayIndexOutOfBounds.

// try — risky code
// catch — handle exception
// finally — always executes (cleanup code)

try {
    int result = 10 / 0;              // ArithmeticException!
    System.out.println("Result: " + result); // skipped
} catch (ArithmeticException e) {
    System.out.println("Caught: " + e.getMessage());
} finally {
    System.out.println("Finally always runs");
}

// Smooth Termination vs Abnormal Termination
// Without handling → Abnormal (program crashes)
// With handling    → Smooth (program continues)

// throw — explicitly throw an exception
public void withdraw(double amt) {
    if (amt > balance) {
        throw new ArithmeticException("Insufficient funds");
    }
    balance -= amt;
}

// throws — declare that method may throw exception
// caller must handle it
public void readFile(String path) throws IOException {
    FileReader fr = new FileReader(path); // may throw IOException
}

// Caller must handle:
try {
    readFile("data.txt");
} catch (IOException e) {
    e.printStackTrace();
}
// Custom Exception — extend Exception or RuntimeException
class AgeException extends Exception {
    AgeException(String msg) {
        super(msg);
    }
}

class Person {
    private int age;

    public void setAge(int age) throws AgeException {
        if(age < 0 || age > 150) {
            throw new AgeException("Invalid age: " + age);
        }
        this.age = age;
    }
}

// Usage
try {
    new Person().setAge(-5);
} catch (AgeException e) {
    System.out.println(e.getMessage()); // Invalid age: -5
}
Invalid age: -5
// Java 7+ Multi-catch — handle multiple exceptions in one block
try {
    int[] arr = new int[5];
    arr[10] = 100;              // ArrayIndexOutOfBoundsException
    String s = null;
    s.length();                  // NullPointerException
} catch (ArrayIndexOutOfBoundsException | NullPointerException e) {
    System.out.println("Caught: " + e.getClass().getSimpleName());
}

// Exception hierarchy — catch child before parent
try {
    // risky code
} catch (ArithmeticException e) {  // specific first
    System.out.println("Arithmetic");
} catch (Exception e) {           // general last
    System.out.println("General");
}
// try-with-resources (Java 7+)
// Auto-closes resources that implement AutoCloseable

try (FileReader fr = new FileReader("file.txt");
     BufferedReader br = new BufferedReader(fr)) {

    String line;
    while((line = br.readLine()) != null) {
        System.out.println(line);
    }
    // fr and br auto-closed here — no finally needed!

} catch (IOException e) {
    e.printStackTrace();
}
